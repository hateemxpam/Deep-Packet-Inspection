cmake_minimum_required(VERSION 3.15)
project(dpi_engine VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(SOURCES
    src/main.cpp
    src/types.cpp
    src/pcap_reader.cpp
    src/packet_parser.cpp
    src/sni_extractor.cpp
    src/flow_tracker.cpp
    src/rule_manager.cpp
    src/reporter.cpp
    src/worker.cpp
    src/mt_engine.cpp
    src/hot_reload.cpp
)

add_executable(dpi_engine ${SOURCES})

target_include_directories(dpi_engine PRIVATE include)

find_package(Threads REQUIRED)
target_link_libraries(dpi_engine PRIVATE Threads::Threads)

target_compile_options(dpi_engine PRIVATE
    $<$<CXX_COMPILER_ID:GNU>:-Wall -Wextra -Wpedantic>
)
